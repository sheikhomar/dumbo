Program

	Number sec
	Number min
	Number hour
	Number remainder

	Write("Enter time in seconds.")
	
	seconds += ReadKey

	hour = seconds / 3600
	remainder = seconds % 3600
	min = remainder / 60
	sec = remainder % 60

	Write("Time is ") Write(hour) Write(" hours, ") Write(min) Write(" minutes and ") Write(sec) Write(" seconds.")
	
End Program
